    private static void f1(String label, int start, int end) {

            if (ii % 3 != 0 && ii % 5 != 0) {
                System.out.print(ii);

                if (ii % 3 == 0) {
                    System.out.print("Fizz");
                }
                if (ii % 5 == 0) {
                    System.out.print("Buzz");
                }
            }
            System.out.println("");
        }
        System.out.println("");
    }
