package kissjgabidolgozat2018.pkg11.pkg12;

import java.util.*;

public class KissJGabiDolgozat20181112 {

    static Scanner sc = new Scanner(System.in);
    static int aa, bb;

    private static void vonalathuz(int n, char c) {
        for (int ii = 0; ii < n; ii++) {
            System.out.print(c);
        }
        System.out.println("");
    }

    private static void f1() {
        String ask0 = "Kérek két egész számot! ",
                ask1 = "Első szám: ",
                ask2 = "Második szám: ";

        System.out.println(ask0);

        System.out.print(ask1);
        aa = sc.nextInt();
        System.out.print(ask2);
        bb = sc.nextInt();
    }

    private static void csere() {
        if (aa > bb) {
            int c = aa;
            aa = bb;
            bb = c;
        }
    }

    private static void sorol() {
        System.out.print(" " + (aa + 1));
        for (int ii = aa + 2; ii < bb; ii++) {
            System.out.print(",  " + ii);
        }
        System.out.println("");
    }

    private static void visszafele() {
        System.out.print(" " + (bb - 1));
        for (int ii = bb - 2; ii > aa; ii--) {
            System.out.print(",  " + ii);
        }
        System.out.println("");
    }

    private static void sorolx(int xx) {
        System.out.print(" " + (aa + 1) * xx);
        for (int ii = aa + 2; ii < bb; ii++) {
            System.out.print(",  " + ii * xx);
        }
        System.out.println("");
    }

    private static void divide(int xx) {
        int counter = 0;
        for (int ii = aa + 1; ii < bb; ii++) {
            if (ii % xx == 0) {
                if (counter > 0) {
                    System.out.print(", ");
                } else {
                    System.out.print(" ");
                }
                System.out.print(ii);
                counter++;
            }
        }
        if (counter == 0) {
            System.out.println("Nincs közte olyan szám, amelyet oszt a " + xx + "!");
        } else {
            System.out.println("");
        }
    }

    public static void main(String[] args) {
        vonalathuz(20, '-');
        f1();
        csere();
        if (bb - aa < 2) {
            System.out.println("Nincs a megadott két szám között egész!");
        } else {
            vonalathuz(8, '*');
            sorol();
            vonalathuz(8, '*');
            visszafele();
            vonalathuz(8, '*');
            sorolx(5);
            vonalathuz(8, '*');
            divide(3);
        }
        vonalathuz(20, '-');
    }
}
